void settei()
{
	printf("-設定画面-\n\n");
	while(1)
	{

		{
			printf("   ■設定一覧\n1:PU変更\n2:ガチャシステム変更\n939:終了\n           -->"); scanf("%d",&k);

		}
		if(k==1)
		{
			//現在表示
			printf("\n==============================\n\n");
			if(gt==1)
			{
				//通常
				printf("現在のPU(通常)\n");
				for(k=0;k<3;k++)
				{
					printf("PU:%d %13s %2s\n",k+1,cr[puc[k]-1],type[put[k]-1]);
				}
			}
			if(gt==2)
			{
				//フェス
				printf("現在のPU(フェス)\n");
				for(k=0;k<5;k++)
				{
					printf("PU:%d %13s %2s\n",k+1,cr[puc2[k]-1],type[put2[k]-1]);
				}
			}
			printf("\n==============================\n");
			//キャラ表示
			printf("\n-キャラクター一覧-\n");
			for(k=0,i=1;k<26;k++,i++)
			{
				printf("%2d: %13s     ",k+1,cr[k]);
				if(i==5)
				{
					printf("\n");
					i=0;
				}
			}
			//タイプ表示
			printf("\n\n-タイプ一覧-\n");
			for(k=0;k<5;k++)
			{
				printf("%2d: %2s     ",k+1,type[k]);
			}
			//変更入力
			if(gt==1)
			{
				//通常
				for(k=0;k<3;k++)
				{
					while(1)
					{
						printf("\n\nPUキャラ:%d\n     -->",k+1); scanf("%d",&puc[k]);
						printf("  タイプ:%d\n     -->",k+1); scanf("%d",&put[k]);
						if(puc[k]>=1&&puc[k]<=26&&put[k]>=1&&put[k]<=5) break;
						else	 printf("範囲外の数値が入力されています。\n");
					}
				}
				printf("\n==============================\n\n");
				for(k=0;k<3;k++)
				{
					printf("PU:%d %13s %2s\n",k+1,cr[puc[k]-1],type[put[k]-1]);
				}
				printf("以下のキャラクターに変更しました。\n\n==============================\n");
			}
			if(gt==2)
			{
				//フェス
				for(k=0;k<5;k++)
				{
					while(1)
					{
						printf("\n\nPUキャラ:%d\n     -->",k+1); scanf("%d",&puc2[k]);
						printf("  タイプ:%d\n     -->",k+1); scanf("%d",&put2[k]);
						if(puc2[k]>=1&&puc2[k]<=26&&put2[k]>=1&&put2[k]<=5) break;
						else	 printf("範囲外の数値が入力されています。\n");
					}
				}
				printf("\n==============================\n\n");
				for(k=0;k<5;k++)
				{
					printf("PU:%d %13s %2s\n",k+1,cr[puc2[k]-1],type[put2[k]-1]);
				}
				printf("以下のキャラクターに変更しました。\n\n==============================\n");
			}
		}
		if(k==2)
		{
			printf("\n-ガチャシステム変更-\n");
			if(gt==1)	printf("   |現在:通常(星4確率3%)|\n");
			else		printf("   |現在:フェス(星4確率6%)|\n");
			while(1)
			{
				printf("1:通常\n2:フェス\n   -->"); scanf("%d",&gt);
				if(gt==1||gt==2) break;
			}
			printf("\n===================================\n\n");
			if(gt==1)printf("ガチャシステムを 通常 にしました。\n");
			else	 printf("ガチャシステムを フェス にしました。\n");
			printf("\n===================================\n");
		}
		if(k==939) break;
	}
	printf("設定を終了します。\n\n");
}
